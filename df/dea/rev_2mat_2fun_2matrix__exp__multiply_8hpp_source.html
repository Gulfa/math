50 inline Eigen::MatrixXd exp_action_chain_vd(double* Ad, double* Bd, const int& n,
51  const int& m,
52  const Eigen::MatrixXd& adjexpAB,
53  const double t) {
54  using Eigen::Map;
55  using Eigen::Matrix;
56  using Eigen::MatrixXd;
57  Eigen::MatrixXd adjexpA = Eigen::MatrixXd::Zero(n, n);
58  Eigen::MatrixXd adjA = Eigen::MatrixXd::Zero(n, n);
59 
60  // TODO(yizhang): a better way such as complex step approximation
61  try {
62  start_nested();
63 
64  adjexpA = adjexpAB * Map<MatrixXd>(Bd, n, m).transpose();
65  Eigen::Matrix<stan::math::var, Eigen::Dynamic, Eigen::Dynamic> Av(n, n);
66  for (int i = 0; i < Av.size(); ++i) {
67  Av(i) = to_var(Ad[i]);
68  }
69  std::vector<stan::math::var> Avec(Av.data(), Av.data() + Av.size());
70  Eigen::Matrix<stan::math::var, Eigen::Dynamic, Eigen::Dynamic> expA
71  = matrix_exp(Av);
72  std::vector<double> g;
73  for (size_type i = 0; i < expA.size(); ++i) {
74  stan::math::set_zero_all_adjoints_nested();
75  expA.coeffRef(i).grad(Avec, g);
76  for (size_type j = 0; j < adjA.size(); ++j) {
77  adjA(j) += adjexpA(i) * g[j];
78  }
79  }
80  } catch (const std::exception& e) {
81  recover_memory_nested();
82  throw;
83  }
84  recover_memory_nested();
